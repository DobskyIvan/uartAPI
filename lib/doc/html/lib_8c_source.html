138eErrorCode uartSend (sUartApi *uartApi, uint8_t *data, uint32_t dataLength){
139 // потенциальный HardFault, нужно обеспечить защиту
141 if (uartApi->temp->txDataLength > 0){
142 return E_COMMERR; // already transmitting
143 }
144
145 uartApi->temp->txDataPointer = data;
146 uartApi->temp->txDataLength = dataLength;
147
148 uartApi->state->txState = STATE_TX_BUSY;
149 uartApi->temp->txDataLength--;
150 uartApi->functions->uartPutByte(*data); // this will call tx interrupt (CBTransmitterEmptyIT())
151
152 return E_NOERR;
153}
